CREATE TABLE SOCIAL_ISOLATION
    (ID        	INT 		NOT NULL 	GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
     CITY  	VARCHAR(50) 	NOT NULL,
     STATE_ID   INT             NOT NULL,
     INDEX 	DOUBLE,
     DATE_WHEN  VARCHAR(10)     NOT NULL,     
     PRIMARY KEY (ID)
);


CREATE TABLE STATE
    (ID        	INT 		NOT NULL 	GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
     NAME  	VARCHAR(30) 	NOT NULL,
     ACRONYM    VARCHAR(2)      NOT NULL,
     PRIMARY KEY (ID)
);

ALTER TABLE SOCIAL_ISOLATION ADD FOREIGN KEY(STATE_ID) REFERENCES STATE(ID);
